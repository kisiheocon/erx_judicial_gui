##Scripted Effects for the GUI
#Effect - Hiring a judge from list
judges_list_button_click_scripted_effect = {
	set_variable = { judicial_gui_com_judge_array^judicial_gui_selected_com_var = judges_list_array^judges_idx }

	if = {
		limit = { check_variable = { judges_list_array^judges_idx = 13 } }
		judge_hire_13_scripted_effect = yes
	}

	remove_from_array = { judges_list_array = judges_list_array^judges_idx }
	set_country_flag = {  flag = judicial_recently_appointed_flag days = 5 value = 1 } #Flag for cooldown
}


#Effect - Firing a judge from list
judge_current_fire_button_click_scripted_effect = {
	add_to_array = { judges_list_array = judicial_gui_com_judge_array^judicial_gui_selected_com_var }

	if = {
		limit = { check_variable = { judicial_gui_com_judge_array^judicial_gui_selected_com_var = 13 } }
		judge_fire_13_scripted_effect = yes
	}

	set_variable = { judicial_gui_com_judge_array^judicial_gui_selected_com_var = 1 }
	set_country_flag = { flag = judicial_recently_fired_flag days = 5 value = 1 } #Flag for cooldown
}
