##Scripted Effects for the GUI
#Effect - Hiring a judge from list
erx_judges_list_button_click_scripted_effect = {
	meta_effect = {
		text = {
			add_ideas = judge_[JUDGE_ID]_hidden_idea
			hidden_effect = { erx_judge_hire_[JUDGE_ID]_scripted_effect = yes } #For unique judges with special effects when hired
		}
		JUDGE_ID = "[?erx_judges_list_array^erx_judges_idx]"
	}
	if = {
		limit = { judges_list_button_click_enabled_scripted_trigger = no }
		custom_effect_tooltip = generic_skip_one_line_tt
	}

	set_variable = { erx_judicial_gui_com_judge_array^erx_judicial_gui_selected_com_var = erx_judges_list_array^erx_judges_idx }
	remove_from_array = { erx_judges_list_array = erx_judges_list_array^erx_judges_idx }
	set_country_flag = { flag = judicial_recently_appointed_flag days = 5 value = 1 } #Flag for cooldown
}

#Effect - Firing a judge from list
erx_judge_current_fire_button_click_scripted_effect = {
	meta_effect = {
		text = {
			if = {
				limit = { NOT = { check_variable = { [JUDGE_ID] = 1 } } }
				remove_ideas = judge_[JUDGE_ID]_hidden_idea
				hidden_effect = { erx_judge_fire_[JUDGE_ID]_scripted_effect = yes } #For unique judges with special effects when fired
				if = {
					limit = { erx_judge_current_fire_button_click_enabled_scripted_trigger = no }
					custom_effect_tooltip = generic_skip_one_line_tt
				}
			}
		}
		JUDGE_ID = "[?erx_judicial_gui_com_judge_array^erx_judicial_gui_selected_com_var]"
	}
	
	add_to_array = { erx_judges_list_array = erx_judicial_gui_com_judge_array^erx_judicial_gui_selected_com_var }
	set_variable = { erx_judicial_gui_com_judge_array^erx_judicial_gui_selected_com_var = 1 }
	set_country_flag = { flag = erx_judicial_recently_fired_flag days = 5 value = 1 } #Flag for cooldown
}
