#Trigger - Hiring a judge from list
erx_judges_list_button_click_enabled_scripted_trigger = {
	if = { #Special for Sierra Commonwealth
		limit = { check_variable = { erx_judicial_gui_selected_com_var = 0 } }
		custom_trigger_tooltip = {
			tooltip = judge_sierra_blocked
			always = no
		}
	}
	else = {
		if = { #Already have a judge in selected Commonwealth
			limit = { NOT = { check_variable = { erx_judicial_gui_com_judge_array^erx_judicial_gui_selected_com_var = 1 } } }
			custom_trigger_tooltip = {
				tooltip = judge_trigger_filled
				always = no
			}
		}
		if = { #Recently appointed a judge in 30 days
			limit = { has_country_flag = erx_judicial_recently_appointed_flag }
			custom_trigger_tooltip = {
				tooltip = judicial_recently_appointed_trigger_tt
				always = no
			}
		}
	}
}

#Trigger - Firing a judge from selected commonwealth
erx_judge_current_fire_button_click_enabled_scripted_trigger = {
	if = { #Special for Sierra Commonwealth
		limit = { check_variable = { erx_judicial_gui_selected_com_var = 0 } }
		custom_trigger_tooltip = {
			tooltip = judge_sierra_blocked
			always = no
		}
	}
	else = {
		if = { #Don't have a judge in selected Commonwealth
			limit = { check_variable = { erx_judicial_gui_com_judge_array^erx_judicial_gui_selected_com_var = 1 } }
			custom_trigger_tooltip = {
				tooltip = judge_trigger_vacant
				always = no
			}
		}
		if = { #Recently fired a judge in 30 days
			limit = { has_country_flag = erx_judicial_recently_fired_flag }
			custom_trigger_tooltip = {
				tooltip = judicial_recently_fired_trigger_tt
				always = no
			}
		}
	}
}
