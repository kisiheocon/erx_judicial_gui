				set_variable = { judicial_gui_com_judge_array^judicial_gui_selected_com_var = judges_list_array^judges_idx }

			}
			judge_current_fire_button_click = {
				add_to_array = { judges_list_array = judicial_gui_com_judge_array^judicial_gui_selected_com_var }
				set_variable = { judicial_gui_com_judge_array^judicial_gui_selected_com_var = 1 }
			}
		}

		triggers = {
			judges_list_portrait_visible = { set_temp_variable = { judges_list_portrait_frame = judges_list_array^judges_idx } }
			judge_current_portrait_visible = { set_temp_variable = { judge_current_portrait_frame = judicial_gui_com_judge_array^judicial_gui_selected_com_var } }
			judges_list_button_click_enabled = {
				NOT = { check_variable = { judicial_gui_selected_com_var = 0 } }
